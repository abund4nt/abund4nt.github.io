gen.py

#!/usr/bin/python

from binascii import hexlify
from gmpy2 import *
import math
import os
import sys

if sys.version_info < (3, 9):
    math.gcd = gcd
    math.lcm = lcm

_DEBUG = False

FLAG  = open('flag.txt').read().strip()
FLAG  = mpz(hexlify(FLAG.encode()), 16)
SEED  = mpz(hexlify(os.urandom(32)).decode(), 16)
STATE = random_state(SEED)

def get_prime(state, bits):
    return next_prime(mpz_urandomb(state, bits) | (1 << (bits - 1)))

def get_smooth_prime(state, bits, smoothness=16):
    p = mpz(2)
    p_factors = [p]
    while p.bit_length() < bits - 2 * smoothness:
        factor = get_prime(state, smoothness)
        p_factors.append(factor)
        p *= factor

    bitcnt = (bits - p.bit_length()) // 2

    while True:
        prime1 = get_prime(state, bitcnt)
        prime2 = get_prime(state, bitcnt)
        tmpp = p * prime1 * prime2
        if tmpp.bit_length() < bits:
            bitcnt += 1
            continue
        if tmpp.bit_length() > bits:
            bitcnt -= 1
            continue
        if is_prime(tmpp + 1):
            p_factors.append(prime1)
            p_factors.append(prime2)
            p = tmpp + 1
            break

    p_factors.sort()

    return (p, p_factors)

e = 0x10001

while True:
    p, p_factors = get_smooth_prime(STATE, 1024, 16)
    if len(p_factors) != len(set(p_factors)):
        continue
    # Smoothness should be different or some might encounter issues.
    q, q_factors = get_smooth_prime(STATE, 1024, 17)
    if len(q_factors) != len(set(q_factors)):
        continue
    factors = p_factors + q_factors
    if e not in factors:
        break

if _DEBUG:
    import sys
    sys.stderr.write(f'p = {p.digits(16)}\n\n')
    sys.stderr.write(f'p_factors = [\n')
    for factor in p_factors:
        sys.stderr.write(f'    {factor.digits(16)},\n')
    sys.stderr.write(f']\n\n')

    sys.stderr.write(f'q = {q.digits(16)}\n\n')
    sys.stderr.write(f'q_factors = [\n')
    for factor in q_factors:
        sys.stderr.write(f'    {factor.digits(16)},\n')
    sys.stderr.write(f']\n\n')

n = p * q

m = math.lcm(p - 1, q - 1)
d = pow(e, -1, m)

c = pow(FLAG, e, n)

print(f'n = {n.digits(16)}')
print(f'c = {c.digits(16)}')

Source code analysis

At first sight the source code is quite dense but I will summarize it and explain it in parts, this in order to give you a better understanding of what it does.

We are dealing with an RSA system that uses soft prime numbers (p and q) to encrypt the flag.

If you are not familiar with RSA you can see this implementation that I made in Python in the simplest possible way so you have a notion before facing the challenge.

https://github.com/luk4spwn/rsa-implementation

We can see in the function get_smooth_prime() how it generates smooth prime numbers.

Definition of a smooth number

In number theory, a smooth number is an integer that can be fully factored into small prime numbers. So a positive January is called B-smooth if none of its prime factors is greater than B. For example the integer 1620 has the prime factorization of 22 x 34 x 5, so 1620 is 5-smooth because none of its prime factors is greater than 5. For a more exact definition you can visit wikipedia.

Solution

Knowing that p and q are smooth prime numbers we must find a way to factor n, researching I came across Pollard’s p -1 algorithm. Pollard’s p-1 algorithm is an integer factoring algorithm that is only suitable for integers with factors of specific types.

The factors it finds are those for which the number preceding the factor p - 1 is a ready power that can be factored into small prime numbers. For further definition visit wikipedia. With the following script we solve the challenge.
